	\begin{frame}{Definição de ER}
		\begin{block}{Definição}
			Dizemos que $R$ é uma {\bf expressão regular} se $R$ for
			\begin{enumerate}
				\item $a$ para algum $a \in \Sigma$;  
				\item $\epsilon$;  
				\item $\emptyset$;  
				\item $(R_1 \cup R_2)$, em que $R_1$ e $R_2$ são expressões regulares;  
				\item $(R_1 \circ R_2)$, em que $R_1$ e $R_2$ são expressões regulares; ou  
				\item $(R_1)^{*}$, em que $R_1$ é expressão regular.
			\end{enumerate}
		\end{block}  
		\begin{alertblock}{Cuidado!!!}
			Não confunda $\epsilon$ com $\emptyset$!!!
		\end{alertblock}
	\end{frame}

	\begin{frame}[shrink]{Exemplos}
		\begin{block}{Exemplo 1.53}
			\begin{itemize}			
				\item $0^* 1 0^*$  
				\item $\Sigma^* 1 \Sigma^*$  
				\item $\Sigma^* 001 \Sigma^*$  
				\item $1^* (01^+)^*$  
				\item $(\Sigma \Sigma)^*$  
				\item $(\Sigma \Sigma \Sigma)^*$  
				\item $01 \cup 10$  
				\item $0\Sigma^* \cup 1\Sigma^* 1 \cup 0 \cup 1$  
				\item $(0 \cup \epsilon) 1^*$  
				\item $(0 \cup \epsilon) (1 \cup \epsilon)$  
				\item $1^* \emptyset$  
				\item $\emptyset^*$
			\end{itemize}
		\end{block} 
	\end{frame}

	\section{Equivalência entre ERs e AFNs}
	\begin{frame}{Teorema}
		\begin{block}{Teorema 1.54}
			Uma linguagem é regular se e somente se alguma expressão regular a descreve. \pause	
			\begin{itemize}
				\item {\bf Lema 1.55: } Uma linguagem é descrita por uma expressão regular, então ela é regular. \pause
				\item {\bf Lema 1.60:} Se uma linguagem é regular, então ela é descrita por uma expressão regular.
			\end{itemize}	
		\end{block} 
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Lema 1.55}
			Uma linguagem é descrita por uma expressão regular, então ela é regular.
		\end{block} \pause
		\begin{block}{Prova}
			Vamos converter $R$ num AFN $N$. Consideramos os seis casos na descrição formal de expressões regulares:
			\begin{itemize} \pause
				\item Três casos básicos;
				\item Três casos gerais.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\item $R = a$ para algum $a$ em $\Sigma$.\\ \pause
				Então $L(R) = \{ a \}$, e o seguinte AFN reconhece $L(R)$. \pause
				\begin{center}
					\includegraphics[width=.3\textwidth]{images/fig-lema-01.png}
				\end{center} \pause
				Formalmente, $N = (\{ q_1, q_2 \}, \Sigma, \delta, q_1, \{ q_2 \})$, em que $\delta$ se divide em dois casos: \pause
				\begin{enumerate}
					\item $\delta(q_1, a) = \{ q_2 \}$
					\item $\delta(r, b) = \emptyset$ (para $r \not= q_1$ ou $b \not= a$)
				\end{enumerate}
			\end{enumerate}		
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\setcounter{enumi}{1}
				\item $R = \epsilon$.\\ \pause
				Então $L(R) = \{ \epsilon \}$, e o seguinte AFN reconhece $L(R)$. \pause
				\begin{center}
					\includegraphics[width=.2\textwidth]{images/fig-lema-02.png}
				\end{center} \pause
				Formalmente, $N = (\{ q_1 \}, \Sigma, \delta, q_1, \{ q_1 \})$, \\em que $\delta(r, b) = \emptyset$ para quaisquer $r$ e $b$.
			\end{enumerate}		
		\end{block}
	\end{frame}
	
	\begin{frame}[shrink]{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\setcounter{enumi}{2}
				\item $R = \emptyset$.\\ \pause
				Então $L(R) = \emptyset$, e o seguinte AFN reconhece $L(R)$. \pause
				\begin{center}
					\includegraphics[width=.2\textwidth]{images/fig-lema-03.png}
				\end{center} \pause
				Formalmente, $N = (\{ q_1 \}, \Sigma, \delta, q_1, \emptyset)$, \\em que $\delta(r, b) = \emptyset$ para quaisquer $r$ e $b$.
			\end{enumerate}		
		\end{block} \pause
		\begin{block}{Casos gerais}
			\begin{enumerate}
				\setcounter{enumi}{3}
				\item $R = R_1 \cup R_2$
				\item $R = R_1 \circ R_2$
				\item $R = R_1^{*}$
			\end{enumerate}
			Para os três casos gerais, utilizamos as provas de que as linguagens regulares são fechadas sob as operações de regulares $\blacksquare$
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema 1.55}
