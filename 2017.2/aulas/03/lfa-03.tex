		\begin{block}{Exemplo: cadeia de entrada {\tt 1101}}
			\begin{enumerate} \pause
				\item Começa no estado $q_1$; \pause 
				\item Lê $1$, segue a transição de $q_1$ para $q_2$; \pause
				\item Lê $1$, segue a transição de $q_2$ para $q_2$; \pause
				\item Lê $0$, segue a transição de $q_2$ para $q_3$; \pause
				\item Lê $1$, segue a transição de $q_3$ para $q_2$; \pause
				\item {\it Aceita}, porque $M_1$ está no estado de aceitação $q_2$ no final da entrada.
			\end{enumerate}
		\end{block}

	\begin{frame}{Definição formal}
		\begin{block}{Por que é importante?}
			\begin{itemize}
				\item Uma definição formal é precisa: \pause
				\begin{itemize}
					\item Os autômatos finitos podem ou não ter 0 estados de aceitação? \pause
					\item Eles devem ter exatamente uma transição saindo de cada estado para cada símbolo de entrada possível? \pause
				\end{itemize}
				\item Uma definição formal provê notação.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Definição formal}
		Um {\bf autômato finito determinístico} (AFD) é uma 5-upla $(Q, \Sigma, \delta, q_0, F)$, de forma que
		
		\begin{enumerate}
			\item $Q$ é um conjunto finito conhecido como os {\bf estados},
			\item $\Sigma$ é um conjunto finito chamado o {\bf alfabeto},
			\item $\delta : Q \times \Sigma \rightarrow Q$ é a {\bf função de transição},
			\item $q_0 \in Q$ é o {\bf estado inicial}, e
			\item $F \subseteq Q$ é o {\bf conjunto de estados de aceitação}.
		\end{enumerate}
	\end{frame}
