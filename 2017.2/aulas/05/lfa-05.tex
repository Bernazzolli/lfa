	\begin{frame}{Notações importantes}
		\begin{block}{Linguagem da máquina $M$}      
			\begin{itemize}
				\item Se $A$ é o conjunto de todas as cadeias que a máquina $M$ aceita, dizemos que $A$ é a {\bf linguagem da máquina $M$};      
				\item $L(M) = A$;      
				\item {\bf $M$ reconhece $A$}.
			\end{itemize}						
		\end{block}      
		\begin{alertblock}{Para evitar mal-entendidos...}
			O termo {\bf aceita} será utilizado para cadeias.\\
			E o termo {\bf reconhece} será utilizado para linguagens.
		\end{alertblock} 
	\end{frame}
	
	\begin{frame}{Notações importantes}
		\begin{exampleblock}{Algo importante...}
			\begin{itemize}
				\item Uma máquina pode aceitar várias cadeias, mas ela sempre reconhece uma {\bf única} linguagem;      
				\item Mesmo se a máquina não aceitar nenhuma cadeia, ela reconhece ainda uma linguagem: $\emptyset$.
			\end{itemize}
		\end{exampleblock}      
		\begin{block}{Em relação a $M_1$...}
			\begin{itemize}
				\item $A = \{ \omega \mbox{ | } \omega$ contém pelo menos um {\tt 1} e um número par de {\tt 0}s segue o último {\tt 1} $\}$;      
				\item $L(M_1) = A$;      
				\item $M_1$ reconhece $A$.
			\end{itemize}
		\end{block}
	\end{frame}

	\section{Definição de Computação e Linguagem Regular}
	\begin{frame}{Computação em um AFD}
		\begin{block}{Definição}
			Seja $M = (Q,\Sigma,\delta, q_0, F)$ um autômato finito e \pause \\
			suponha que $\omega = \omega_1 \omega_2 \ldots \omega_n$ seja uma cadeia \pause \\
			em que cada $\omega_i$ é um membro do alfabeto $\Sigma$ ($1 \leq i \leq n$). \pause \\Então $M$ {\bf aceita} $\omega$ se uma sequência de estados $r_0, r_1, \ldots, r_n$ em $Q$ existe satisfazendo três condições: \pause
			\begin{enumerate}
				\item $r_0 = q_0$;
				\item $\delta(r_{i}, \omega_{i+1}) = r_{i+1}$;
				\item $r_n \in F$.
			\end{enumerate}
		\end{block} \pause
		\begin{block}{Corolário}
			$M$ reconhece a linguagem $A$, se $A = \{ \omega$  | $M$ aceita $\omega \}$.
		\end{block}
	\end{frame}
	
	\begin{frame}{Linguagem regular}
		\begin{block}{Definição 1.16}
			Uma linguagem é chamada de uma {\bf linguagem regular} se algum autômato finito a reconhece.
		\end{block} \pause
		\begin{block}{Exemplos de computação}
			Máquina $M_5$ e a cadeia $\omega =$ {\tt 10$\langle$RESET$\rangle$22$\langle$RESET$\rangle$012}:
			\begin{itemize}
				\item $q_0, q_1, q_1, q_0, q_2, q_1, q_0, q_0, q_1, q_0$ \pause
				\item $L(M_5) = \{ \omega$ | a soma dos símbolos em $\omega$ é {\tt 0} módulo {\tt 3}, exceto que {\tt $\langle$RESET$\rangle$} retorna o contador para {\tt 0}$\}$
			\end{itemize}
		\end{block}
	\end{frame}
