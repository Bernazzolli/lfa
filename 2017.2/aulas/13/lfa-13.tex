	\subsection{Fecho em Linguagens Regulares (cont.)}
	\begin{frame}[shrink]{Operações regulares}
		\begin{block}{Teorema 1.25}
			A classe de linguagens regulares é fechada sob a operação de união.
		\end{block}
		\begin{block}{Prova}
			Sejam $A$ e $B$ duas linguagens regulares. Se $A$ e $B$ são regulares, então existem dois AFDs $M_A = (Q_A, \Sigma_A, \delta_A, q_A, F_A)$ e $M_B = (Q_B, \Sigma_B, \delta_B, q_B, F_B)$ que as reconhecem, respectivamente. Como passo auxiliar, iremos construir o AFDs estendidos $M_A' = (Q_A', \Sigma_A', \delta_A', q_A', F_A')$ e $M_B' = (Q_B', \Sigma_B', \delta_B', q_B', F_B')$ dos AFDs $M_A$ e $M_B$, respectivamente. Um AFD estendido $O$ é um AFD equivalente a um dado AFD $P$ de forma que $\Sigma_P \subset \Sigma O$. Desta forma, temos:
		\end{block}
	\end{frame}
	
	\begin{frame}{Operações regulares}
		\begin{block}{Prova (cont.)}
			Elementos de $M_A'$:
			\begin{itemize}
				\item $Q_A' = Q_A \cup \{ q_{fugaA}\}$;
				\item $\Sigma_A' = \Sigma_A \cup \Sigma_B$;
				\item $\delta_A'(q,a) = \left\{\begin{array}{rl}
					\delta_A(q,a), & \text{se } a \in \Sigma_A\\
					q_{fugaA}, & \text{caso contrário}
				\end{array} \right.$\\
				em que $q \in Q_B'$ e $a \in \Sigma_B'$;
				\item $q_A' = q_A$;
				\item $F_A' = F_A$.
			\end{itemize}
		\end{block}
	\end{frame}

	\begin{frame}{Operações regulares}
		\begin{block}{Prova (cont.)}
			Elementos de $M_B'$:
			\begin{itemize}
				\item $Q_B' = Q_B \cup \{ q_{fugaB}\}$;
				\item $\Sigma_B' = \Sigma_A \cup \Sigma_B$;
				\item $\delta_B'(q,a) = \left\{\begin{array}{rl}
				\delta_B(q,a), & \text{se } a \in \Sigma_B\\
				q_{fugaB}, & \text{caso contrário}
				\end{array} \right.$\\
				em que $q \in Q_B'$ e $a \in \Sigma_B'$;
				\item $q_B' = q_B$;
				\item $F_B' = F_B$.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}[shrink]{Operações regulares}
		\begin{block}{Prova (cont.)}
			De posse de $M_A'$ e $M_B'$, será construído o AFD $M_{A \cup B} = (Q, \Sigma, \delta, q_0, F)$ que reconhece $A \cup B$:
			
			Elementos de $M_{A \cup B}$:
			\begin{itemize}
				\item $Q = Q_A' \times Q_B'$;
				\item $\Sigma = \Sigma_A'$;
				\item \begin{center}
				 		$\delta((x,y),a) = (\delta_A'(x,a), \delta_B'(y,a))$\\
				 		em que $(x,y) \in Q$ e $a \in \Sigma$;
					  \end{center}
				
				\item $q_0 = (q_A', q_B')$;
				\item $F = \{ (x, y) \in Q$ | $x \in F_A'$ ou $y \in F_B' \}$.	
			\end{itemize}
			Assim, como foi possível construir $M_{A \cup B}$, podemos dizer que a classe de linguagens regulares é fechada sob a operação de união $\blacksquare$
		\end{block}
	\end{frame}

	\begin{frame}{Computação em um AFN}
		\begin{block}{Definição}
			Seja $N = (Q,\Sigma,\delta, q_0, F)$ um AFN e \pause \\
			suponha que $\omega = \omega_1 \omega_2 \ldots, \omega_n$ seja uma cadeia \pause \\
			em que cada $\omega_i$ é um membro do alfabeto $\Sigma_{\epsilon}$ ($1 \leq i \leq n$). \pause \\Então $N$ {\bf aceita} $\omega$ se uma sequência de estados $r_0, r_1, \ldots, r_n$ em $Q$ existe satisfazendo três condições: \pause
			\begin{enumerate}
				\item $r_0 = q_0$;
				\item $r_{i+1} \in \delta(r_i, \omega_{i+1})$ (para $i=0, \ldots , n-1$); e
				\item $r_n \in F$.
			\end{enumerate}
		\end{block} \pause
		\begin{block}{Corolário}
			$N$ reconhece a linguagem $A$, se $A = \{ \omega$  | $N$ aceita $\omega \}$.
		\end{block}
	\end{frame}
