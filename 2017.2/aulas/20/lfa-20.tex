	\begin{frame}{Teorema}
		\begin{block}{Teorema 1.54}
			Uma linguagem é regular se e somente se alguma expressão regular a descreve.  	
			\begin{itemize}
				\item {\bf Lema 1.55: } Uma linguagem é descrita por uma expressão regular, então ela é regular.  
				\item {\bf Lema 1.60:} Se uma linguagem é regular, então ela é descrita por uma expressão regular.
			\end{itemize}	
		\end{block} 
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Lema 1.55}
			Uma linguagem é descrita por uma expressão regular, então ela é regular.
		\end{block}  
		\begin{block}{Prova}
			Vamos converter $R$ num AFN $N$. Consideramos os seis casos na descrição formal de expressões regulares:
			\begin{itemize}  
				\item Três casos básicos;
				\item Três casos gerais.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\item $R = a$ para algum $a$ em $\Sigma$.\\  
				Então $L(R) = \{ a \}$, e o seguinte AFN reconhece $L(R)$.  
				\begin{center}
					\includegraphics[width=.3\textwidth]{images/fig-lema-01.png}
				\end{center}  
				Formalmente, $N = (\{ q_1, q_2 \}, \Sigma, \delta, q_1, \{ q_2 \})$, em que $\delta$ se divide em dois casos:  
				\begin{enumerate}
					\item $\delta(q_1, a) = \{ q_2 \}$
					\item $\delta(r, b) = \emptyset$ (para $r \not= q_1$ ou $b \not= a$)
				\end{enumerate}
			\end{enumerate}		
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\setcounter{enumi}{1}
				\item $R = \epsilon$.\\  
				Então $L(R) = \{ \epsilon \}$, e o seguinte AFN reconhece $L(R)$.  
				\begin{center}
					\includegraphics[width=.2\textwidth]{images/fig-lema-02.png}
				\end{center}  
				Formalmente, $N = (\{ q_1 \}, \Sigma, \delta, q_1, \{ q_1 \})$, \\em que $\delta(r, b) = \emptyset$ para quaisquer $r$ e $b$.
			\end{enumerate}		
		\end{block}
	\end{frame}
	
	\begin{frame}[shrink]{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\setcounter{enumi}{2}

				Então $L(R) = \emptyset$, e o seguinte AFN reconhece $L(R)$.  
				\begin{center}

		\end{block}  
		\begin{block}{Casos gerais}
			\begin{enumerate}
				\setcounter{enumi}{3}
				\item $R = R_1 \cup R_2$
				\item $R = R_1 \circ R_2$
				\item $R = R_1^{*}$
			\end{enumerate}
			Para os três casos gerais, utilizamos as provas de que as linguagens regulares são fechadas sob as operações de regulares $\blacksquare$
		\end{block}
	\end{frame}

	\begin{frame}
		\begin{block}{Pergunta 18.1}
			Se $\Sigma$ é o alfabeto da linguagem, a expressão regular simplificada $\Sigma^k$ gera a linguagem composta por...
		\end{block}
		\begin{enumerate}[(A)]
			\item todas as $k$ cadeias com símbolos de $\Sigma$
			\item todas as cadeias com símbolos $k$ em $\Sigma$
			\item todas as cadeias de comprimento $k$ em $\Sigma$
			\item todas as cadeias de comprimento até $k$ em $\Sigma$
		\end{enumerate}
	\end{frame}
	
	\begin{frame}
		\begin{block}{Pergunta 18.2}
			Indique a afirmação \underline{\bf incorreta} sobre os autômatos finitos não-determinísticos generalizado (AFNG).
		\end{block}
		\begin{enumerate}[(A)]
			\item o seu estado inicial deve ter setas de transição indo para todos os demais estados.
			\item tem um único estado final que recebe setas de transição vindo de todos os demais estados.
			\item as suas setas de transição podem ser uma expressão regular, lendo assim blocos de símbolos da entrada dada.
			\item deve existir sempre setas de transição indo de qualquer estado para qualquer outro estado, incluindo laços.
		\end{enumerate}
	\end{frame}

	\begin{frame}
		\begin{block}{Pergunta 18.3}
			Um dos processos possíveis de converter um AFN em uma expressão regular equivalente é através do uso de um AFNG. \\ \underline{\bf Não} faz parte deste processo...
		\end{block}
		\begin{enumerate}[(A)]
			\item transformar o AFN em um AFNG equivalente
			\item garantir que cada AFNG tem setas de transições $\epsilon$ \\saindo do estado inicial para todos os outros demais estados.
			\item converter cada AFNG gerado de $k$ estados em \\um outro AFNG equivalente com $k-1$ estados (sendo $k > 2$).
			\item obter a expressão regular através da única seta de transição do AFNG, quando houver apenas dois estados.
		\end{enumerate}
	\end{frame}

		\section{Gramáticas Livre-de-Contexto}
	\begin{frame}{O que são as GLCs?}
		\begin{itemize}
			\item Método mais poderoso de descrição de linguagens; \pause
			\item Linguagens como $\{ 0^n 1^n | n \geq 0 \}$ podem agora ser descritas; \pause
			\item Descrição de linguagens com estruturas recursivas; \pause
			\item GLCs geram {\it Linguagens Livres-de-Contexto} (LLCs).
		\end{itemize} \pause
		\begin{block}{Aplicações...}
			\begin{itemize}
				\item Especificação e compilação de linguagens de programação.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Exemplos...}
		\begin{block}{}
			\begin{itemize}
				\item[] $A \rightarrow 0 A 1$
				\item[] $A \rightarrow B$
				\item[] $B \rightarrow \#$ 
			\end{itemize} 
		\end{block} \pause
		\begin{block}{Uma gramática consiste de...}
			\begin{itemize}
				\item Regras de substituição (também chamadas de {\bf produções}); \pause
				\item Símbolos que são chamados de {\bf variáveis}; \pause
				\item Símbolos que são chamados de {\bf terminais}; \pause
				\item Uma variável é designada como {\bf variável inicial}.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Como utilizar uma gramática?}	
		\begin{itemize}
			\item {\bf Escreva a variável inicial.} Ela é a variável no lado esquerdo da primeira regra, a menos que especificado em contrário; \pause
			\item {\bf Encontre uma variável que esteja escrita e uma regra que comece com essa variável.} Substitua a variável escrita pelo lado direito dessa regra; \pause
			\item Repita o passo 2 até que não reste nenhuma variável.
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Exemplo}
		\begin{block}{$G_1$ gera a cadeia $000\# 111$}

	\begin{frame}{Outras características importantes}
		\begin{itemize}
			\item O conjunto de todas as cadeias geradas por uma GLC constitui a {\bf linguagem da gramática}; \pause
			\item $L(G_1) = \{ 0^n \# 1^n | n \geq 0 \}$ \pause
			\item Qualquer linguagem que possa ser gerada por uma GLC é chamada de {\bf linguagem livre-do-contexto (LLC)}; \pause
			\item Podemos abreviar várias regras em apenas uma só como $A \rightarrow 0 A 1$ e $A \rightarrow B$ em \\
			$A \rightarrow 0 A 1 | B$.
		\end{itemize}
	\end{frame}
