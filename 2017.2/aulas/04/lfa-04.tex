			\begin{enumerate}     
				\item Começa no estado $q_1$;      
				\item Lê $1$, segue a transição de $q_1$ para $q_2$;     
				\item Lê $1$, segue a transição de $q_2$ para $q_2$;     
				\item Lê $0$, segue a transição de $q_2$ para $q_3$;     
				\item Lê $1$, segue a transição de $q_3$ para $q_2$;

			\end{enumerate}
		\end{block}

	\begin{frame}{Definição formal}
		\begin{block}{Por que é importante?}
			\begin{itemize}
				\item Uma definição formal é precisa:     
				\begin{itemize}
					\item Os autômatos finitos podem ou não ter 0 estados de aceitação?     
					\item Eles devem ter exatamente uma transição saindo de cada estado para cada símbolo de entrada possível?     
				\end{itemize}
				\item Uma definição formal provê notação.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Definição formal}
		Um {\bf autômato finito determinístico} (AFD) é uma 5-upla $(Q, \Sigma, \delta, q_0, F)$, de forma que
		
		\begin{enumerate}
			\item $Q$ é um conjunto finito conhecido como os {\bf estados},
			\item $\Sigma$ é um conjunto finito chamado o {\bf alfabeto},
			\item $\delta : Q \times \Sigma \rightarrow Q$ é a {\bf função de transição},
			\item $q_0 \in Q$ é o {\bf estado inicial}, e
			\item $F \subseteq Q$ é o {\bf conjunto de estados de aceitação}.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Notações importantes}
		\begin{block}{Linguagem da máquina $M$} \pause
			\begin{itemize}
				\item Se $A$ é o conjunto de todas as cadeias que a máquina $M$ aceita, dizemos que $A$ é a {\bf linguagem da máquina $M$}; \pause
				\item $L(M) = A$; \pause
				\item {\bf $M$ reconhece $A$}.
			\end{itemize}						
		\end{block} \pause
		\begin{alertblock}{Para evitar mal-entendidos...}
			O termo {\bf aceita} será utilizado para cadeias.\\
			E o termo {\bf reconhece} será utilizado para linguagens.
		\end{alertblock} 
	\end{frame}
	
	\begin{frame}{Notações importantes}
		\begin{exampleblock}{Algo importante...}
			\begin{itemize}
				\item Uma máquina pode aceitar várias cadeias, mas ela sempre reconhece uma {\bf única} linguagem; \pause
				\item Mesmo se a máquina não aceitar nenhuma cadeia, ela reconhece ainda uma linguagem: $\emptyset$.
			\end{itemize}
		\end{exampleblock} \pause
		\begin{block}{Em relação a $M_1$...}
			\begin{itemize}
				\item $A = \{ \omega \mbox{ | } \omega$ contém pelo menos um {\tt 1} e um número par de {\tt 0}s segue o último {\tt 1} $\}$; \pause
				\item $L(M_1) = A$; \pause
				\item $M_1$ reconhece $A$.
			\end{itemize}
		\end{block}
	\end{frame}

			L(M_5) = \{ \omega \mbox{ | } \left\{ \begin{array}{rl}
			\left(  \sum\limits_{i=1}^{n} \omega_i \right) \equiv_3 0, 	
			& \mbox{em que } \langle RESET \rangle \not= \omega_j\\
			& \mbox{para } 1 \leq j \leq n, \\			
			\left(  \sum\limits_{i=j+1}^{n} \omega_i \right) \equiv_3 0, 	
			& \mbox{em que } \omega_j \mbox{ é o último }\\
			& \langle RESET \rangle \mbox{ em } \omega \\
			\end{array}	
			\right. \}
			$$
		\end{block}
