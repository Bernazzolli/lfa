	\subsection{Equivalência de AFNs e AFDs}
	\begin{frame}{Computação em um AFN}
		\begin{block}{Definição}
			Seja $N = (Q,\Sigma,\delta, q_0, F)$ um AFN e   \\
			suponha que $\omega = \omega_1 \omega_2 \ldots, \omega_n$ seja uma cadeia   \\
			em que cada $\omega_i$ é um membro do alfabeto $\Sigma_{\epsilon}$ ($1 \leq i \leq n$).   \\Então $N$ {\bf aceita} $\omega$ se uma sequência de estados $r_0, r_1, \ldots, r_n$ em $Q$ existe satisfazendo três condições:  
			\begin{enumerate}
				\item $r_0 = q_0$;
				\item $r_{i+1} \in \delta(r_i, \omega_{i+1})$ (para $i=0, \ldots , n-1$); e
				\item $r_n \in F$.
			\end{enumerate}
		\end{block}  
		\begin{block}{Corolário}
			$N$ reconhece a linguagem $A$, se $A = \{ \omega$  | $N$ aceita $\omega \}$.
		\end{block}
	\end{frame}
	
	\begin{frame}{Equivalência de AFNs e AFDs}
		\begin{block}{Teorema 1.39}
			Todo autômato finito não-determinístico tem um autômato finito determinístico equivalente.
		\end{block}  
		\begin{block}{Ideia da prova}
			\begin{itemize}
				\item Se uma linguagem é reconhecida por um AFN, então temos de mostrar a existência de um AFD que também a reconhece;  
				\item Converter um AFN num AFD equivalente que o simule;  
				\item Se $k$ é o número de estados do AFN, então ele tem $2^k$ subconjuntos de estados;  
				\item Portanto o AFD equivalente terá $2^k$ estados.
			\end{itemize}
		\end{block}
	\end{frame}

			A classe de linguagens regulares é fechada sob a operação de união.

	\begin{frame}{Fecho sob união}
		\begin{block}{Estrutura básica da prova}
			Suponha que $N_1 = (Q_1, \Sigma, \delta_1, q_1, F_1)$ reconheça $A_1$ e que \\
			$N_2 = (Q_2, \Sigma, \delta_2, q_2, F_2)$ reconheça $A_2$. \\  
			Construa $N = (Q, \Sigma, \delta, q_0, F)$ para reconhecer $A_1 \cup A_2$.  
			\begin{itemize}
				\item $Q = Q_1 \cup Q_2 \cup \{q_0\}$;  
				\item $\Sigma$ é o alfabeto da linguagem;  
				\item $\delta(q,a) = \left\{\begin{array}{cl}  
				
					\delta_1(q,a), & 
					\text{se } q \in Q_1\\  
					
					\delta_2(q,a), & 
					\text{se } q \in Q_2\\  
					
					\{q_1, q_2\}, & 
					\text{se } q = q_0 \text{ e } a = \epsilon\\  
				
					\emptyset, & 
					\text{se } q=q_0 \text{ e } a \not= \epsilon.
				\end{array} \right.$\\  
				em que $q \in Q$ e $a \in \Sigma_{\epsilon}$;  
				\item $q_0$ é o estado inicial;  
				\item $F = F_1 \cup F_2$. 
			\end{itemize}
		\end{block}
	\end{frame}

	\begin{frame}{Fecho sob concatenação}
		\begin{block}{Estrutura básica da prova}
			Suponha que $N_1 = (Q_1, \Sigma, \delta_1, q_1, F_1)$ reconheça $A_1$ e que \\
			$N_2 = (Q_2, \Sigma, \delta_2, q_2, F_2)$ reconheça $A_2$. \\  
			Construa $N = (Q, \Sigma, \delta, q_1, F_2)$ para reconhecer $A_1 \circ A_2$.  
			\begin{itemize}
				\item $Q = Q_1 \cup Q_2$;  
				\item $\Sigma$ é o alfabeto da linguagem;  
				\item $\delta(q,a) = \left\{\begin{array}{cl}  
				
					\delta_1(q,a), & 
					\text{se } q \in Q_1 \text{ e } q \not\in F_1\\  
					
					\delta_1(q,a), & 
					\text{se } q \in F_1 \text{ e } a \not= \epsilon\\  
					
					\delta_1(q,a) \cup \{q_2\}, & 
					\text{se } q \in F_1 \text{ e } a = \epsilon\\  
					
					\delta_2(q,a), & 
					\text{se } q\in Q_2.
				\end{array} \right.$\\  
				em que $q \in Q$ e $a \in \Sigma$;  
				\item $q_1$ é o estado inicial;  
				\item $F_2$ é o conjunto de estados finais. 
			\end{itemize}
		\end{block}

	\section{Expressão Regular}	
	\begin{frame}{Exemplos}

		\end{block}
	\end{frame}		
	
	\begin{frame}{Definição de ER}
		\begin{block}{Definição}
			Dizemos que $R$ é uma {\bf expressão regular} se $R$ for
			\begin{enumerate}
				\item $a$ para algum $a \in \Sigma$; \pause
				\item $\epsilon$; \pause
				\item $\emptyset$; \pause
				\item $(R_1 \cup R_2)$, em que $R_1$ e $R_2$ são expressões regulares; \pause
				\item $(R_1 \circ R_2)$, em que $R_1$ e $R_2$ são expressões regulares; ou \pause
				\item $(R_1)^{*}$, em que $R_1$ é expressão regular.
			\end{enumerate}
		\end{block} \pause
		\begin{alertblock}{Cuidado!!!}
			Não confunda $\epsilon$ com $\emptyset$!!!
		\end{alertblock}
	\end{frame}

				\item {\bf Lema 1.55: } Uma linguagem é descrita por uma expressão regular, então ela é regular. \pause

	\begin{frame}{Prova do Lema}
		\begin{block}{Lema 1.55}
			Uma linguagem é descrita por uma expressão regular, então ela é regular.
		\end{block} \pause
		\begin{block}{Prova}
			Vamos converter $R$ num AFN $N$. Consideramos os seis casos na descrição formal de expressões regulares:
			\begin{itemize} \pause
				\item Três casos básicos;
				\item Três casos gerais.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\item $R = a$ para algum $a$ em $\Sigma$.\\ \pause
				Então $L(R) = \{ a \}$, e o seguinte AFN reconhece $L(R)$. \pause
				\begin{center}
					\includegraphics[width=.3\textwidth]{images/fig-lema-01.png}
				\end{center} \pause
				Formalmente, $N = (\{ q_1, q_2 \}, \Sigma, \delta, q_1, \{ q_2 \})$, em que $\delta$ se divide em dois casos: \pause
				\begin{enumerate}
					\item $\delta(q_1, a) = \{ q_2 \}$
					\item $\delta(r, b) = \emptyset$ (para $r \not= q_1$ ou $b \not= a$)
				\end{enumerate}
			\end{enumerate}		
		\end{block}
	\end{frame}
	
	\begin{frame}{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}
				\setcounter{enumi}{1}
				\item $R = \epsilon$.\\ \pause
				Então $L(R) = \{ \epsilon \}$, e o seguinte AFN reconhece $L(R)$. \pause
				\begin{center}
					\includegraphics[width=.2\textwidth]{images/fig-lema-02.png}
				\end{center} \pause
				Formalmente, $N = (\{ q_1 \}, \Sigma, \delta, q_1, \{ q_1 \})$, \\em que $\delta(r, b) = \emptyset$ para quaisquer $r$ e $b$.
			\end{enumerate}		
		\end{block}
	\end{frame}
	
	\begin{frame}[shrink]{Prova do Lema}
		\begin{block}{Prova do Lema 1.55}
			\begin{enumerate}

		\end{block} \pause
		\begin{block}{Casos gerais}

				\item $R = R_1 \circ R_2$
				\item $R = R_1^{*}$
			\end{enumerate}
			Para os três casos gerais, utilizamos as provas de que as linguagens regulares são fechadas sob as operações de regulares $\blacksquare$
		\end{block}
	\end{frame}
