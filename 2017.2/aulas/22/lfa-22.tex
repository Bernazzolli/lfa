	\section{Revisão}
	\subsection{Gramáticas Livre-de-Contexto}
	\begin{frame}{O que são as GLCs?}
		\begin{itemize}
			\item Método mais poderoso de descrição de linguagens; \pause
			\item Linguagens como $\{ 0^n 1^n | n \geq 0 \}$ podem agora ser descritas; \pause
			\item Descrição de linguagens com estruturas recursivas; \pause
			\item GLCs geram {\it Linguagens Livres-de-Contexto} (LLCs).
		\end{itemize} \pause
		\begin{block}{Aplicações...}
			\begin{itemize}
				\item Especificação e compilação de linguagens de programação.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Exemplos...}
		\begin{block}{}
			\begin{itemize}
				\item[] $A \rightarrow 0 A 1$
				\item[] $A \rightarrow B$
				\item[] $B \rightarrow \#$ 
			\end{itemize} 
		\end{block} \pause
		\begin{block}{Uma gramática consiste de...}
			\begin{itemize}
				\item Regras de substituição (também chamadas de {\bf produções}); \pause
				\item Símbolos que são chamados de {\bf variáveis}; \pause
				\item Símbolos que são chamados de {\bf terminais}; \pause
				\item Uma variável é designada como {\bf variável inicial}.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Como utilizar uma gramática?}	
		\begin{itemize}
			\item {\bf Escreva a variável inicial.} Ela é a variável no lado esquerdo da primeira regra, a menos que especificado em contrário; \pause
			\item {\bf Encontre uma variável que esteja escrita e uma regra que comece com essa variável.} Substitua a variável escrita pelo lado direito dessa regra; \pause
			\item Repita o passo 2 até que não reste nenhuma variável.
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Exemplo}
		\begin{block}{$G_1$ gera a cadeia $000\# 111$}

	\begin{frame}{Outras características importantes}
		\begin{itemize}
			\item O conjunto de todas as cadeias geradas por uma GLC constitui a {\bf linguagem da gramática}; \pause
			\item $L(G_1) = \{ 0^n \# 1^n | n \geq 0 \}$ \pause
			\item Qualquer linguagem que possa ser gerada por uma GLC é chamada de {\bf linguagem livre-do-contexto (LLC)}; \pause
			\item Podemos abreviar várias regras em apenas uma só como $A \rightarrow 0 A 1$ e $A \rightarrow B$ em \\
			$A \rightarrow 0 A 1 | B$.
		\end{itemize}
	\end{frame}

	\section{Gramáticas Livre-de-Contexto}
	\begin{frame}{Exemplo}
		\begin{block}{$G_1$}
			\begin{itemize}
				\item[] $A \rightarrow 0 A 1$
				\item[] $A \rightarrow B$
				\item[] $B \rightarrow \#$ 
			\end{itemize} 
		\end{block} \pause
		\begin{block}{Descrição formal}
			$G_1 = ( V, \Sigma, R, S )$ de forma que:
			\begin{itemize}
				\item $V = \{ A, B \}$;
				\item $\Sigma = \{ 0, 1, \# \}$;
				\item $R = \{ A \rightarrow 0 A 1, \mbox{ } A \rightarrow B, \mbox{ } B \rightarrow \# \}$;
				\item $S = A$.
			\end{itemize}
		\end{block}
	\end{frame}
	
	\begin{frame}{Exemplo}
		\begin{block}{$G_3$}
			$G_3 = ( \{ S \}, \{ a, b \}, R, S )$. O conjunto de regras $R$ é:
			\begin{itemize}
				\item[] $S \rightarrow a S b \mbox{ | } SS \mbox{ | } \epsilon$
			\end{itemize} 
		\end{block} \pause
		\begin{block}{$G_4$}
			$G_4 = ( V, \Sigma, R, S)$, de forma que:
			\begin{itemize}
				\item[] $V = \{ \langle EXPR \rangle, \langle TERM \rangle, \langle FACTOR \rangle \}$;
				\item[] $\Sigma = \{ a, +, \times, (, ) \}$;
				\item[] $R$ é o conjunto de regras:
				\begin{itemize}
					\item[] $ \langle EXPR \rangle \rightarrow \langle EXPR \rangle + \langle TERM \rangle \mbox{ | } \langle TERM \rangle$
					\item[] $ \langle TERM \rangle \rightarrow \langle TERM \rangle \times \langle FACTOR \rangle \mbox{ | } \langle FACTOR \rangle$
					\item[] $ \langle FACTOR \rangle \rightarrow ( \langle EXPR \rangle ) \mbox{ | } a $
				\end{itemize} 
				\item[] $S = \langle EXPR \rangle$.
			\end{itemize}
		\end{block}
	\end{frame}

	\begin{frame}{Ambiguidade}
		\begin{block}{Definição 2.7}
			Uma cadeia $\omega$ é derivada {\bf ambiguamente} na gramática livre-do-contexto $G$ se ela tem duas ou mais derivações mais à esquerda diferentes. A gramática $G$ é {\bf ambígua} se ela gera alguma cadeia ambiguamente.
		\end{block} \pause
		\begin{block}{}
			Existem linguagens que são {\bf inerentemente ambíguas}!
		\end{block}
	\end{frame}
	
	\subsection{Forma Normal de Chomsky}
	\begin{frame}[shrink]{Forma Normal de Chomsky}
		\begin{block}{Por quê utilizá-la?}
			\begin{itemize}
				\item É uma forma simplificada de escrever uma GLC;
				\item Facilita a construção de algoritmos para GLC.
			\end{itemize}
		\end{block} \pause
		\begin{block}{Definição 2.8}
			Uma gramática livre-do-contexto está na {\bf forma normal de Chomsky} se toda regra é da forma
			\begin{itemize}
				\item[] $A \rightarrow BC$
				\item[] $A \rightarrow a$
			\end{itemize}
			em que $a$ é qualquer terminal e $A$, $B$ e $C$ são quaisquer variáveis - exceto que $B$ e $C$ não podem ser a variável inicial.
		\end{block} \pause
		\begin{exampleblock}{Adicionalmente...}
			Permite-se a regra $S \rightarrow \epsilon$, em que $S$ é a variável inicial.
		\end{exampleblock}
	\end{frame}
	
	\begin{frame}{Forma Normal de Chomsky}
		\begin{block}{Teorema 2.9}
			Qualquer linguagem livre-de-contexto é gerada por uma gramática livre-do-contexto na forma normal de Chomsky.
		\end{block}
	\end{frame}
